```java
public class Main {
    public static void main(String[] args) {
        int num = 42;
        double pi = 3.14159;
        String str = "Hello, World!";
        char ch = 'A';

        // 输出整数
        System.out.printf("整数: %d%n", num);

        // 输出浮点数，保留两位小数
        System.out.printf("浮点数: %.2f%n", pi);

        // 输出字符串
        System.out.printf("字符串: %s%n", str);

        // 输出字符
        System.out.printf("字符: %c%n", ch);
    }
}
```
